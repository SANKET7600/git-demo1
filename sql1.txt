										-- AGGREGATE FUNCTIONS --
-- SQL aggregate function are used to perform the calculation on multiple rows of a single column
-- and return a single value.
-- there are different type of aggregate function are:-
-- (1) Count()	-> count() is used to find total value of the columns.
-- (2) Sum() 	-> sum() is used to find addition value of the columns.
-- (3) Avg() 	-> avg() is used to find average value of the columns.
-- (4) Min() 	-> min() is used to find lowest value of the columns.
-- (5) Max() 	-> max() is used to find highest value of the columns.


-- LIKE opertator will use to get records based on pattern. --
-- WHERE CustomerName LIKE 'a%'   Finds any values that start with "a"
-- WHERE CustomerName LIKE '%a'   Finds any values that end with "a"
-- WHERE CustomerName LIKE '%or%' Finds any values that have "or" in any position
-- WHERE CustomerName LIKE '_r%'  Finds any values that have "r" in the second position
-- WHERE CustomerName LIKE 'a_%'  Finds any values that start with "a" and are at least 2 character in length
-- WHERE CustomerName LIKE 'a__%' Finds any values that start with "a" and are at least 3 character in length
-- WHERE CustomerName LIKE 'a%o'  Finds any values that start with "a" and ends with "o"

	-- what is maximum salary in sales department?

	-- what is average salary in finance department?

	-- what is maximum salary in IT deparment?

	-- find unique department from employee table
select distinct(dept_name) from employee;

	-- find total unique department count
select count(distinct(dept_name)) from employee;

	-- what is the minimum salary of each department?
select dept_name, min(emp_salary) from employee group by dept_name;

	-- what is the maximum salary of each department?
select dept_name, max(emp_salary) from employee group by dept_name;

	-- Find average of each department
select dept_name, avg(emp_salary) from employee group by dept_name;

	-- find how many employee in each department
select dept_name, count(emp_name) from employee group by dept_name;

	-- find total pay to each employee in each department
select dept_name, sum(emp_salary) from employee group by dept_name;

	-- find total pay to each employee in each department Order by sum of employee salary
select dept_name, sum(emp_salary) from employee group by dept_name order by sum(emp_salary) desc;

	-- find employee in each department order by employee salary
select dept_name from employee group by dept_name order by sum(emp_salary) desc;

	-- find average salary into asc and desc format
select dept_name, avg(emp_salary) from employee group by dept_name order by avg(emp_salary) desc;

	-- find maximum salary into each department order by asc and desc format
select dept_name, max(emp_salary) from employee group by dept_name order by max(emp_salary) desc;

	-- find minimum salary into each department order by asc and desc format
select dept_name, min(emp_salary) from employee group by dept_name order by min(emp_salary) desc;

	-- Having Clause
	-- find department name maximum salary bigger than average salary
select dept_name, max(emp_salary) from employee group by dept_name having avg(emp_salary)<max(emp_salary);

	-- find department name more than 2 employee 
select dept_name, count(emp_id) from employee group by dept_name having count(emp_id) > 2;

	-- find department name less than 2 employee 
select dept_name, count(emp_id) from employee group by dept_name having count(emp_id) <= 2;
